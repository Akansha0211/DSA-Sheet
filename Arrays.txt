(Approach 2)
public static String reverseWord(String str)
    {
        // Reverse the string str
        char arr[] = str.toCharArray();
        int start = 0;
        int end = arr.length-1;
        while(start<end){
            char temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
        return new String(arr);
    }
